    public class SelectionBean implements Serializable {
        @GetSet
        private List<String> selectedFruits;
        @Getter
        private final List<String> fruitOptions = Arrays.asList("Apple", "Banana", "Orange", "Grape", "Mango");
    }
    <h:form>
        <h:outputLabel value="Select your favorite fruits:" for="fruits" />
        <h:selectManyListbox id="fruits" value="#{selectionBean.selectedFruits}" size="5">
            <f:selectItems value="#{selectionBean.fruitOptions}" />
        </h:selectManyListbox>
        <br/>
        <h:commandButton value="Submit" />
        <h:outputText value="You selected: #{selectionBean.selectedFruits}" />
    </h:form>
