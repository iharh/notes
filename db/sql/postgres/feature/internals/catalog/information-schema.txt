pg_catalog.
    pg_aggregate


SELECT 
    p.proname AS aggregate_name,
    p.pronamespace::regnamespace AS schema_name,
    a.aggfnoid::regproc AS function_name
FROM 
    pg_aggregate a
JOIN 
    pg_proc p ON a.aggfnoid = p.oid
    ;
WHERE 
    p.proname = 'json_arrayagg';
