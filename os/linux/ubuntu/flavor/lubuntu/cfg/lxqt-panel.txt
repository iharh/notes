~/.config/lxqt/
    globalkeyshortcuts.conf
        C-A-b   browser
        C-A-Del qps (task manager)
        C-A-l   lock screen
        C-A-t   qterminal
        M-0     task 0
        M-1     task 1
        ...
        M-9     task 9
        M-d     show/hide desktop
        M-F1    switch to desktop 1
        M-F2    switch to desktop 2
        ...
        M-F12   switch to desktop 12
        M-r     lxqt-runner
        Print   screenshot
    lxqt-runner.conf
        https://github.com/lxqt/lxqt-runner
