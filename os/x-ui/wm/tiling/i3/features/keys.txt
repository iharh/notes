mod-CR
    terminal
mod-d
    dmenu

mod-shift-e
    exit i3 session
mod-shift-c
    reload i3 config
mod-shift-r
    restart i3 in place

mod-<num>
    go to workspace number <num>
mod-n
    go to next workspace
mod-p
    go to previous workspace
mod-shift-<num>
    move the current window to workspace <num>

mod-shift-q
    close a title
mod-space
    toggle focus on floating tiles
mod-shift-space
    toggle floating
mod-r
    toggle resize
mod-f
    toggle fullscreen
